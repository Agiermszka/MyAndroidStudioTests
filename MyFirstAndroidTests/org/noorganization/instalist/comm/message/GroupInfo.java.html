23       private Integer mId;
24

27           return mReadableId;
28       }
29   
30       @JsonProperty("accesskey")
31       public void setReadableId(String _readableId) {
32           mReadableId = _readableId;
33       }
34   
35       public GroupInfo withReadableId(String _id) {
36           mReadableId = _id;
37           return this;
38       }
39   
40       @JsonProperty("id")
41       public Integer getId() {
42           return mId;
43       }
44   
45       @JsonProperty("id")
46       public void setId(Integer _id) {
47           mId = _id;
48       }
49   
50       public GroupInfo withId(Integer _id) {
51           setId(_id);
52           return this;
53       }
54   }
